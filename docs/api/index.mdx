### With Configuration Panel

For a complete animation interface:

```tsx
import { AnimatedContainer } from '@/components/animated-container';
import { ConfigPanel } from '@/components/config-panel';
import { useAnimationConfig } from '@/hooks/useAnimationConfig';

function AnimationPlayground() {
  const { animationConfig, setAnimationConfig } = useAnimationConfig();

  return (
    <div className="flex gap-4">
      <ConfigPanel
        initialConfig={animationConfig}
        onConfigChange={setAnimationConfig}
      />
      <AnimatedContainer config={animationConfig}>
        <div>Preview your animation here</div>
      </AnimatedContainer>
    </div>
  );
}
```

## ðŸ”§ Configuration

### Default Animation Config

```typescript
const defaultConfig: AnimationConfig = {
  type: 'fade',
  duration: 0.5,
  easing: 'ease-out',
  delay: 0,
  name: 'My Animation',
};
```

### Supported Animation Types

- `fade` - Opacity transitions
- `slide` - Transform-based sliding
- `scale` - Size scaling animations
- `rotate` - Rotation transforms
- `bounce` - Spring-based bouncing

### Easing Functions

- `ease`, `ease-in`, `ease-out`, `ease-in-out`
- `linear`
- `cubic-bezier(n,n,n,n)` - Custom cubic bezier curves

### Custom Animations

Create custom animation configurations:

```typescript
const customConfig: AnimationConfig = {

  duration: 1.2,
  easing: 'cubic-bezier(0.4, 0, 0.2, 1)',
  delay: 0.3,
  name: 'Custom Slide',
  // Additional custom properties...
};
```

### Programmatic Control

Control animations programmatically:

```typescript
import { generateAnimationCSS } from '@/utils/animations';

const cssString = generateAnimationCSS(animationConfig);
// Use the generated CSS for custom implementations
```
