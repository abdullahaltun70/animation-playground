# EasingFunction Type

The `EasingFunction` type defines the timing curves and easing functions available for animations in the Animation Playground.

## Import

```typescript
import type { EasingFunction } from '@/types/animation';
```

## Type Definition

```typescript

  // CSS Predefined
```

## CSS Predefined Easings

### `linear`

Constant speed throughout the animation.

```typescript
const linearConfig: AnimationConfig = {
  type: 'slide',
  duration: 1000,
  easing: 'linear', // No acceleration or deceleration
};
```

**Cubic-bezier equivalent:** `cubic-bezier(0, 0, 1, 1)`

**Use cases:**

- Continuous rotations
- Loading indicators
- Mechanical movements

### `ease`

Default easing with slow start and end, faster middle.

```typescript
const easeConfig: AnimationConfig = {
  type: 'fade',
  duration: 800,
  easing: 'ease', // Standard natural motion
};
```

**Cubic-bezier equivalent:** `cubic-bezier(0.25, 0.1, 0.25, 1)`

**Use cases:**

- General purpose animations
- Default choice for most effects

### `ease-in`

Slow start, accelerating towards the end.

```typescript
const easeInConfig: AnimationConfig = {
  type: 'scale',
  duration: 600,
  easing: 'ease-in', // Gentle start, strong finish
};
```

**Cubic-bezier equivalent:** `cubic-bezier(0.42, 0, 1, 1)`

**Use cases:**

- Elements leaving the screen
- Dramatic entrances building up speed

### `ease-out`

Fast start, decelerating towards the end.

```typescript
const easeOutConfig: AnimationConfig = {
  type: 'bounce',
  duration: 1000,
  easing: 'ease-out', // Strong start, gentle finish
};
```

**Cubic-bezier equivalent:** `cubic-bezier(0, 0, 0.58, 1)`

**Use cases:**

- Elements entering the screen
- Settling into final position

### `ease-in-out`

Slow start and end, fastest in the middle.

```typescript
const easeInOutConfig: AnimationConfig = {
  type: 'slide',
  duration: 1200,
  easing: 'ease-in-out', // Smooth acceleration and deceleration
};
```

**Cubic-bezier equivalent:** `cubic-bezier(0.42, 0, 0.58, 1)`

**Use cases:**

- Smooth transitions
- Professional-looking animations

## Custom Cubic Bezier

### Creating Custom Curves

```typescript
const customEasing: AnimationConfig = {
  type: 'flip',
  duration: 800,
  easing: 'cubic-bezier(0.68, -0.55, 0.265, 1.55)', // Overshoot effect
};
```

**Parameters:** `cubic-bezier(x1, y1, x2, y2)`

- `x1, y1`: First control point (0-1)
- `x2, y2`: Second control point (0-1)

### Popular Custom Curves

```typescript
const popularEasings = {
  // Gentle overshoot
  softBack: 'cubic-bezier(0.68, -0.55, 0.265, 1.55)',

  // Strong overshoot
  strongBack: 'cubic-bezier(0.68, -0.8, 0.265, 1.8)',

  // Smooth acceleration
  smoothAccel: 'cubic-bezier(0.4, 0, 0.2, 1)',

  // Dramatic ease-in
  dramaticIn: 'cubic-bezier(0.7, 0, 0.84, 0)',

  // Snappy
  snappy: 'cubic-bezier(0.4, 0, 0.6, 1)',
};
```

## Named Easing Functions

### Sine Easings

Smooth, gentle curves based on sine function.

```typescript
// Sine easing options
const sineEasings = {
  easeInSine: 'cubic-bezier(0.12, 0, 0.39, 0)',
  easeOutSine: 'cubic-bezier(0.61, 1, 0.88, 1)',
  easeInOutSine: 'cubic-bezier(0.37, 0, 0.63, 1)',
};

const sineExample: AnimationConfig = {
  type: 'fade',
  duration: 1000,
  easing: 'easeInOutSine', // Very smooth fade
};
```

### Quad Easings

Quadratic acceleration curves.

```typescript
const quadExample: AnimationConfig = {
  type: 'slide',
  duration: 800,
  easing: 'easeOutQuad', // Quick start, gentle stop
};
```

### Cubic Easings

More pronounced acceleration than quad.

```typescript
const cubicExample: AnimationConfig = {
  type: 'scale',
  duration: 600,
  easing: 'easeInCubic', // Slow build to dramatic finish
};
```

### Back Easings

Overshoot the target before settling.

```typescript
const backExample: AnimationConfig = {
  type: 'bounce',
  duration: 1000,
  easing: 'easeOutBack', // Overshoot then settle
};
```

### Elastic Easings

Spring-like oscillation effect.

```typescript
const elasticExample: AnimationConfig = {
  type: 'scale',
  duration: 1500,
  easing: 'easeOutElastic', // Bouncy spring effect
};
```

### Bounce Easings

Multiple bounces before settling.

```typescript
const bounceExample: AnimationConfig = {
  type: 'slide',
  duration: 1200,
  easing: 'easeOutBounce', // Multiple bounces at end
};
```

## Usage Examples

### Easing Selection Component

```typescript
import { EasingFunction } from '@/types/animation';

interface EasingSelectProps {
  value: EasingFunction;
  onChange: (easing: EasingFunction) => void;
}

function EasingSelect({ value, onChange }: EasingSelectProps) {
  const easingGroups = {
    'CSS Standard': ['linear', 'ease', 'ease-in', 'ease-out', 'ease-in-out'],
    'Sine': ['easeInSine', 'easeOutSine', 'easeInOutSine'],
    'Quad': ['easeInQuad', 'easeOutQuad', 'easeInOutQuad'],
    'Cubic': ['easeInCubic', 'easeOutCubic', 'easeInOutCubic'],
    'Back': ['easeInBack', 'easeOutBack', 'easeInOutBack'],
    'Elastic': ['easeInElastic', 'easeOutElastic', 'easeInOutElastic'],
    'Bounce': ['easeInBounce', 'easeOutBounce', 'easeInOutBounce']
  };

  return (
    <select value={value} onChange={(e) => onChange(e.target.value as EasingFunction)}>
      {Object.entries(easingGroups).map(([group, easings]) => (
        <optgroup key={group} label={group}>
          {easings.map(easing => (
            <option key={easing} value={easing}>
              {easing}
            </option>
          ))}
        </optgroup>
      ))}
    </select>
  );
}
```

```typescript
}
```

### Performance Considerations

```typescript
// Efficient for simple animations
const efficientEasings: EasingFunction[] = [
  'linear',
  'ease',
  'ease-in',
  'ease-out',
  'ease-in-out',
];

// More complex calculations (use sparingly)
const complexEasings: EasingFunction[] = [
  'easeInElastic',
  'easeOutElastic',
  'easeInBounce',
  'easeOutBounce',
];

function getRecommendedEasing(animationType: AnimationType): EasingFunction {
  const recommendations: Record<AnimationType, EasingFunction> = {
    fade: 'ease-in-out',
    slide: 'ease-out',
    scale: 'ease-out',
    rotate: 'linear',
    bounce: 'easeOutBounce',
    // ... more recommendations
  };

  return recommendations[animationType] || 'ease-in-out';
}
```

## Best Practices

### Choosing the Right Easing

```typescript
const easingGuidelines = {
  // Element entering screen
  entrance: ['ease-out', 'easeOutBack', 'easeOutBounce'],

  // Element leaving screen
  exit: ['ease-in', 'easeInBack'],

  // Two-way transitions
  bidirectional: ['ease-in-out', 'easeInOutSine', 'easeInOutQuad'],

  // Attention-grabbing
  attention: ['easeOutElastic', 'easeOutBounce'],

  // Subtle, professional
  subtle: ['ease', 'easeInOutSine', 'easeInOutQuad'],

  // Continuous motion
  continuous: ['linear'],
};
```

### Duration and Easing Pairing

```typescript
const durationEasingPairs = {
  // Quick interactions (< 300ms)
  quick: ['ease-out', 'easeOutQuad'],

  // Standard animations (300-800ms)
  standard: ['ease-in-out', 'easeInOutSine'],

  // Longer animations (> 800ms)
  long: ['ease', 'easeInOutCubic', 'easeOutBack'],

  // Micro-interactions (< 200ms)
  micro: ['ease-out', 'linear'],
};
```
